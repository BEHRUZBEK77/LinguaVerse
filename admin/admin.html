Firestore Security Rules
"Missing or insufficient permissions" xatosi chiqsa — quyidagi rules ni Firebase Console ga qo'ying:
console.firebase.google.com → Firestore Database → Rules → Publish
Firebase Console Rules
rules_version = '2'; service cloud.firestore { match /databases/{database}/documents { function isAdmin() { return request.auth != null && request.auth.token.email in [ 'owner@gmail.uz', 'linguaverse.admin@gmail.com' ]; } function isOwner(uid) { return request.auth != null && request.auth.uid == uid; } match /{document=**} { allow read, write: if isAdmin(); } match /users/{userId} { allow read, write: if isOwner(userId); } match /orders/{orderId} { allow create: if request.auth != null; allow read, update: if request.auth != null && ( request.auth.uid == resource.data.uid || request.auth.uid == resource.data.userId ); } match /purchaseRequests/{docId} { allow create: if request.auth != null; } match /announcements/{docId} { allow read: if request.auth != null; } } }
