🔐 Firestore Security Rules — MUHIM!
Agar "Missing or insufficient permissions" xatosi chiqsa, quyidagi rules ni Firebase Console ga qo'ying:
console.firebase.google.com → Firestore Database → Rules tab → Tahrirlash → Publish
🔗 Firebase Console → Rules ni ochish
rules_version = '2'; service cloud.firestore { match /databases/{database}/documents { // ── Admin emaillarni o'zgartiring ── function isAdmin() { return request.auth != null && request.auth.token.email in [ 'owner@gmail.uz', 'linguaverse.admin@gmail.com' ]; } function isOwner(uid) { return request.auth != null && request.auth.uid == uid; } // Admin hamma narsaga to'liq kirish match /{document=**} { allow read, write: if isAdmin(); } // Foydalanuvchilar o'z ma'lumotlarini o'qiy/yoza oladi match /users/{userId} { allow read, write: if isOwner(userId); } // Buyurtmalar: yaratish + o'z buyurtmasini ko'rish match /orders/{orderId} { allow create: if request.auth != null; allow read, update: if request.auth != null && ( request.auth.uid == resource.data.uid || request.auth.uid == resource.data.userId ); } // To'lov so'rovlari: yaratish match /purchaseRequests/{docId} { allow create: if request.auth != null; } // E'lonlar: o'qish match /announcements/{docId} { allow read: if request.auth != null; } } }
